import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

data = {'Name': ['Alice', 'Bob', 'Charlie', 'David', 'Eve', 'Frank'], 'Department': ['Marketing', 'HR', 'Marketing', 'Finance', 'Marketing', 'IT']}
df = pd.DataFrame(data)
marketing_count = (df["Department"] == "Marketing").sum()
print("Marketing count:", marketing_count)
sns.countplot(x="Department", data=df, palette="Set2")
